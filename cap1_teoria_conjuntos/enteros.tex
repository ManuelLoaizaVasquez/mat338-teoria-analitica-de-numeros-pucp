\begin{proposition}
    Las operaciones suma
    $$[(a_0, b_0)] + [(a_1, b_1)] = [(a_0 + a_1, b_0 + b_1)]$$
    y producto
    $$[(a_0, b_0)][(a_1, b_1)] = [(a_0 a_1 + b_0 b_1, a_0 b_1 + a_1 b_0)]$$
    est\'an bien definidos.
\end{proposition}
